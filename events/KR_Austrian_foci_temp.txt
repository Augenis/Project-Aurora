    is_triggered_only = yes

		AUS = {
			transfer_state = 69
			transfer_state = 74
			transfer_state = 786
		}
	}

    option = {
        name = aus.9001.a
		ai_chance = { factor = 30 }
		AUS = {
			annex_country = {

				transfer_troops = yes
			}

		}
		set_global_flag = Dualism_Bohemia
    }

    option = {
        name = aus.9001.b
		ai_chance = { factor = 70 }
		set_global_flag = Austrian_Civil_War
		AUS = { 
			end_puppet = BOM
		}
		BOM = {
			declare_war_on = {
				target = AUS
				type = annex_everything
			}
		}
		AUS = {
			if = {
				limit = {
					exists = yes
				}
				country_event = aus.9003
			}
		}
		GAL = {
			if = {
				limit = {
					exists = yes
				}
				country_event = aus.9004
			}
		}
		ILL = {
			if = {
				limit = {
					exists = yes
				}
				country_event = aus.9005
			}
		}
		HUN = {
			if = {
				limit = {
					exists = yes
				}
				country_event = aus.9006
			}
		}
    }
}
