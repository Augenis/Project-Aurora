#ACW
news_event = {
	id = worldnews.1
	title = worldnews.1.t
	desc = worldnews.1.d
	picture = GFX_news_event_032
	
	major = yes
	
	trigger = {
		country_exists = CSA
		country_exists = USA
		country_exists = TEX
		USA = {
			AND = { has_war_with = CSA 
				has_war_with = TEX  }}
		NOT = { tag = USA }
		NOT = { tag = CSA }
		NOT = { tag = TEX }
		NOT = { has_global_flag = ACW_has_happened }
		}
		
	immediate = { set_global_flag = ACW_has_happened }

    option = {
		name = worldnews.1.a
	}
}
