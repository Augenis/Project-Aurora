## Selling the railway - Germany
country_event = {
	id = ethiopia.2
	title = etheopia.2.t
	desc = ethiopia.2.d
	fire_only_once = yes

	trigger = {
		tag = GER
		date > 1936.6.8
		NOT = {
		GER = { has_war_with = ETH }
		GER = { has_war_with = MAF }
		ETH = { has_war_with = MAF }
		}
	}

	mean_time_to_happen = {
		days = 5
	}

	option = { # Keep it running!
		name = ethiopia.2.a
		ai_chance = { factor = 20 }
		add_political_power = -60
		add_opinion_modifier = { target = ETH modifier = GER_railway_run }
		add_opinion_modifier = { target = MAF modifier = GER_railway_run }
		hidden_effect = {
			country_event = { id = ethipoia.5 days = 360 }
		}
		custom_effect_tooltip = ethiopia.railway.bankrupcy
	}

	option = { # Sell the railway!
		name = ethiopia.2.b
		ai_chance = { factor 80 }
		add_opinion_modifier = { target = ETH modifier = GER_railway_sold }
		add_opinion_modifier = { target = MAF modifier = GER_railway_sold }
		hidden_effect = {
			ETH = { country_event = { id = ethipoia.3 days = 1 } }
			MAF = { country_event = { tid = ethipoia.4 days = 1 } }
		}
	}
}
