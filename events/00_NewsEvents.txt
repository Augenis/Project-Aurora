	is_triggered_only = yes

	option = {
		name = worldnews.2.a

			NOT = {
				TAG = GER
				TAG = FRA
        TAG = SWI
			}
		}
	}
	option = {
		name = worldnews.2.b
		trigger = { TAG = SWI }
	}

		name = worldnews.2.c
		trigger = { TAG = FRA }
	}

	}
