	fire_only_once = yes

		tag = DEH
		has_war_with = AFG
		has_country_flag = DEH_afghan_war
		AFG = {
			owns_state = 442 #Peshawar
			owns_state = 445 #Quetta
			surrender_progress > 0.30
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = { # take cores
		name = delhi.5.a
		ai_chance = {
			factor = 50
		}
		AFG = {
			remove_state_core = 442 #Peshawar
			remove_state_core = 445 #Quetta
			white_peace_with_ROOT = yes
		}
		DEH = {
			transfer_state = 442 #Peshawar
			transfer_state = 445 #Quetta
		}
		hidden_effect = {
			AFG = { country_event = { id = afg.1 } }
		}
	}
	option = { # puppet
		name = delhi.5.b
		ai_chance = {
			factor = 49
		}
