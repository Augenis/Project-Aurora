	id = lithuania.5

	picture = black_monday
	
	trigger = {
		tag = LIT
		GER = {
			has_idea = GER_black_monday
		}
	}
	
	mean_time_to_happen = { days = 45 }
	
	fire_only_once = yes
	
	option = {
		name = lithuania.5.a
		add_political_power = -50
		add_ideas = LIT_black_monday
		hidden_effect = {
			country_event = { id = lithuania.6 days = 30 }

	}

###REVOLUCION!
country_event = { 
	id = lithuania.11
	title = lithuania.11.t
	desc = lithuania.11.d
	picture = LIT_revolution

	fire_only_once = yes
	
	trigger ={ 
		tag = LIT
		NOT = { has_country_flag = totalist_banned }
		OR = {
			at_war_with = FRA
			at_war_with = ENG
			at_war_with = SOV
		}
		NOT = { has_government = totalist }
		is_in_faction_with = GER
		is_puppet_of = GER
		OR = { 
			AND = {
			64 = { is_controlled_by = GER }
			NOT = { 64 = { is_owned_by = Ger } }
			}
			any_war_score < -20
		}
	}
		
	mean_time_to_happen = {days = 10}
		
	option = {
		name = lithuania.11.a
		ai_chance = { factor = 75 }
		add_political_power = -200
		add_popularity = { ideology = totalist popularity = 0.3 }
		GER = { remove_from_faction = LIT }
		GER = { end_puppet = LIT }
		set_politics = { ruling_party = totalist elections_allowed = no }
		if = { limit = { at_war_with = FRA } white_peace = FRA
		if = { limit = { at_war_with = ENG } white_peace = ENG
		if = { limit = { at_war_with = SOV } white_peace = SOV
	}
	option = {
		name = lithuania.11.b
		ai_chance = { factor = 25 }
		add_political_power = -300
	}

}
