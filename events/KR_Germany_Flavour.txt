# Begin

news_event = {
	id = gerflavour.3
	title = gerflavour.3.t
	desc = gerflavour.3.d
	picture = GER_hoffmann_richthoffen
	
	trigger = {
		tag = GER
		OR = {
			controls_state = 546
			any_country = {
				limit = {
					is_puppet_of = GER
				}
				controls_state = 546
			}
		}
		NOT = { 
			has_country_flag = Manfred_Freiherr_von_Richthofen_dead 
		}
		date > 1939.6.9
	}
	
	mean_time_to_happen = { days = 2 }
	
	fire_only_once = yes
	
	option = {
		name = gerflavour.3.a 
		random_list = {
			50 = { GER = { news_event = { id = gerflavour.4 days = 4 } } }
			50 = { GER = { news_event = { id = gerflavour.5 days = 4 } } }
		}
	}
}
