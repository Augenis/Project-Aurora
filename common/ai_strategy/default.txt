entente_france = { 
	enable = {
		OR = { #Either
			AND = { #Be cananda and be a factiona leader
				tag = CAN
				is_faction_leader = yes
			}
			AND = { # Be in a faction with Canada. Where Canada is the faction leader
				is_in_faction = yes
				any_allied_country = {
					tag = CAN
					is_faction_leader = yes
				}
			}
		}
		FRA = {
			OR = { #France must have a syndiclist government
				has_government = totalist
				has_government = syndicalist
				has_government = radical_socialist
			}
		}
		CAN = {
			OR = { #Canada must not have a syndiclist government
				has_government = social_democrat
				has_government = social_liberal
				has_government = market_liberal
				has_government = social_conservative	
				has_government = authoritarian_democrat	
				has_government = paternal_autocrat 
				has_government = national_populist
			}
		}
	}
	abort = {
		OR = { #If one of the above conditions is no longer met, abort the strategy
			NOT = {
				OR = {
					AND = {
						tag = CAN
						is_faction_leader = yes
					}
					AND = {
						is_in_faction = yes
						any_allied_country = {
							tag = CAN
							is_faction_leader = yes
						}
					}
				}
			}
			FRA = {
				NOT = {
					OR = {
						has_government = totalist
						has_government = syndicalist
						has_government = radical_socialist
					}
				}
			}
			CAN = {
				NOT = {
					OR = {
						has_government = social_democrat
						has_government = social_liberal
						has_government = market_liberal
						has_government = social_conservative	
						has_government = authoritarian_democrat	
						has_government = paternal_autocrat 
						has_government = national_populist
					}
				}
			}
		}
	}
	
	ai_strategy = {
		type = contain
		id = "FRA"			
		value = 200
	}
	ai_strategy = {
		type = conquer
		id = "FRA"			
		value = 200
	}
	ai_strategy = {
		type = antagonize
		id = "FRA"			
		value = 200
	}
}

spanish_civil_war-kingdom_support= {
	enable = {
		is_subject = no #cannot be a subject
		has_war = no #must be at peace
		any_owned_state = { #ensure that the country is not land locked
			is_coastal = yes 
		} 
		NOT = { tag = POR }
		NOT = { tag = AUS }
		NOT = { tag = ITA }
		NOT = { tag = IRE }
		NOT = { capital_scope = { is_on_continent = asia } }
		OR = { #must have a democratically leaning government
			has_government = social_democrat
			has_government = social_liberal
			has_government = market_liberal
			has_government = social_conservative	
			has_government = authoritarian_democrat	
			tag = RUS
		}
		OR = { #Helper must be a major power or in a faction
			is_major = yes
			is_in_faction = yes
			tag = AUS
			tag = ITA
			tag = MEX
			tag = BRA
			tag = POR
			tag = IRE
		}
		SPR = { #Spanish Kingdom must be at war
			has_war = yes
		}
		OR = { #Either spanish revolter must exist
			CAR = { #Cartlist spain exists
				exists = yes
			}
			CNT = { #CNT Spain Exists
				exists = yes
			}
		}
	}
	abort = { #stop sending troops; Stop if any of the above conditions are no longer met
		OR = {
			is_subject = yes 
			has_war = yes
			tag = POR
			tag = AUS
			tag = ITA
			tag = IRE
			capital_scope = { is_on_continent = asia }
			NOT = {
				OR = {
					has_government = social_democrat
					has_government = social_liberal
					has_government = market_liberal
					has_government = social_conservative	
					has_government = authoritarian_democrat	
					tag = RUS
				}
			}

					tag = IRE
				}
			}
			SPR = { 
				has_war = no
			}
			NOT = {
				OR = {
					CAR = { 
						exists = yes
					}
					CNT = { 
						exists = yes
					}
				}
			}
		}
	}	
	ai_strategy = {
		type = send_volunteers_desire
